SELECT CITY,NVL(STATE_PROVINCE,'NO TIENE') FROM LOCATIONS;
SELECT FIRST_NAME||' '||LAST_NAME AS NAME,NVL(SALARY+SALARY*COMMISSION_PCT,SALARY) AS SUELDO FROM EMPLOYEES; -- PODEMOS RECREAR EL NVL2 DE ESTA FORMA.
SELECT FIRST_NAME AS NAME, NVL2(COMMISSION_PCT,SALARY+SALARY*COMMISSION_PCT,SALARY) AS SUELDO FROM EMPLOYEES; --ESTE PRIMERO REVISA SI ES NULO EN LUGAR DE CAPTURAR LA CONDICION.

SELECT DEPARTMENT_NAME,NVL(MANAGER_ID,-1) AS MANAGER_ID FROM DEPARTMENTS;

SELECT NULLIF(CITY,STATE_PROVINCE) AS REGRESA_NULO,COALESCE(NULLIF(CITY,STATE_PROVINCE),CITY) AS REGRESA_CIUDAD FROM LOCATIONS;

SELECT 
CASE DEPARTMENT_ID
    WHEN 50 THEN 'TRANSPORTE'
    WHEN 90 THEN 'DIRECCION'
    ELSE 'OTRO DEPARTAMENTO'
    END AS TIPO_TRABAJO, FIRST_NAME,DEPARTMENT_ID
FROM EMPLOYEES;

SELECT
CASE 
WHEN COUNTRY_ID = 'US' OR COUNTRY_ID = 'CA' THEN 'AMERICA DEL NORTE'
WHEN COUNTRY_ID IN ('CH','UK','DE','IT') THEN 'EUROPA'
WHEN COUNTRY_ID = 'BR' THEN 'AMERICA DEL SUR'
ELSE 'OTRA ZONA'
END AS REGION,COUNTRY_ID,COUNTRY_NAME,REGION_ID
FROM COUNTRIES;

SELECT DECODE(DEPARTMENT_ID,50,'TRANSPORTE',90,'DIRECCION','OTRO DEPARTAMENTO') AS TIPO_TRABAJO,FIRST_NAME,DEPARTMENT_ID FROM EMPLOYEES;
