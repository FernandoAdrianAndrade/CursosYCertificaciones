SELECT * FROM LOCATIONS WHERE STATE_PROVINCE IS NULL;

SELECT FIRST_NAME ||' '||LAST_NAME AS NAME, SALARY, commission_pct, SALARY + (SALARY * COMMISSION_PCT) AS SUELDO_TOTAL FROM EMPLOYEES WHERE COMMISSION_PCT IS NOT NULL; 

SELECT FIRST_NAME ||' '||LAST_NAME AS NAME, HIRE_DATE, JOB_ID,SALARY FROM EMPLOYEES WHERE JOB_ID LIKE 'IT_PROG' AND SALARY < 6000;

SELECT * FROM EMPLOYEES WHERE (DEPARTMENT_ID = 50 OR DEPARTMENT_ID = 80) AND FIRST_NAME LIKE 'S%' AND SALARY > 3000;
